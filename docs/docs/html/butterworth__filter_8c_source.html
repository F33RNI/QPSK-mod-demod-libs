43butter_filter_s *butter_filter_init(uint8_t pass_type,
44 float sample_rate,
45 uint8_t order,
46 float cutoff_frequency,
47 float resonance)
48{
49 // Allocate struct
50 butter_filter_s *filter = malloc(sizeof(butter_filter_s));
51
52 // Initialize basic fields
53 filter->pass_type = pass_type;
54 filter->sample_rate = sample_rate;
55 filter->order = order;
56 filter->cutoff_frequency = cutoff_frequency;
57 filter->resonance = resonance;
58
59 // Initialize filtering constants
60 if (pass_type == FILTER_TYPE_LOWPASS)
61 {
62 float c = 1.f / tanf(M_PI * cutoff_frequency / sample_rate);
63 filter->a1 = 1.f / (1.f + resonance * c + c * c);
64 filter->a2 = 2.f * filter->a1;
65 filter->a3 = filter->a1;
66 filter->b1 = 2.f * (1.f - c * c) * filter->a1;
67 filter->b2 = (1.f - resonance * c + c * c) * filter->a1;
68 }
69 else
70 {
71 float c = tanf(M_PI * cutoff_frequency / sample_rate);
72 filter->a1 = 1.f / (1.f + resonance * c + c * c);
73 filter->a2 = -2.f * filter->a1;
74 filter->a3 = filter->a1;
75 filter->b1 = 2.f * (c * c - 1.f) * filter->a1;
76 filter->b2 = (1.f - resonance * c + c * c) * filter->a1;
77 }
78
79 // Allocate and initialize input histories
80 filter->input_histories = malloc(order * sizeof(float *));
81 for (uint8_t i = 0; i < order; i++)
82 {
83 filter->input_histories[i] = malloc(2 * sizeof(float));
84 for (uint8_t j = 0; j < 2; j++)
85 filter->input_histories[i][j] = 0.f;
86 }
87
88 // Allocate and initialize output histories
89 filter->output_histories = malloc(order * sizeof(float *));
90 for (uint8_t i = 0; i < order; i++)
91 {
92 filter->output_histories[i] = malloc(3 * sizeof(float));
93 for (uint8_t j = 0; j < 3; j++)
94 filter->output_histories[i][j] = 0.f;
95 }
96
97 return filter;
98}

107float butter_filter_filter(butter_filter_s *filter, float input_value)
108{
109 // Filter same value multiple times
110 filter->filtered_value_temp = input_value;
111 for (uint8_t order_i = 0; order_i < filter->order; order_i++)
112 {
113 filter->output_temp = filter->a1 * filter->filtered_value_temp +
114 filter->a2 * filter->input_histories[order_i][0] +
115 filter->a3 * filter->input_histories[order_i][1] -
116 filter->b1 * filter->output_histories[order_i][0] -
117 filter->b2 * filter->output_histories[order_i][1];
118
119 filter->input_histories[order_i][1] = filter->input_histories[order_i][0];
120 filter->input_histories[order_i][0] = filter->filtered_value_temp;
121
122 filter->output_histories[order_i][2] = filter->output_histories[order_i][1];
123 filter->output_histories[order_i][1] = filter->output_histories[order_i][0];
124 filter->output_histories[order_i][0] = filter->output_temp;
125
126 // Save for next order
127 filter->filtered_value_temp = filter->output_temp;
128 }
129
130 return filter->filtered_value_temp;
131}
