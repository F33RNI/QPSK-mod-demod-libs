61qpsk_demodulator_s *qpsk_demodulator_init(float sample_rate,
62 float lo_frequency,
63 float bandwidth,
64 uint16_t halfcycles_per_symbol,
65 float carrier_start_threshold,
66 float carrier_lost_threshold,
67 float pll_lock_threshold,
68 float pll_lock_time)
69{
70 // Allocate struct
71 qpsk_demodulator_s *qpsk_demodulator = malloc(sizeof(qpsk_demodulator_s));
72
73 // Initialize basic fields
74 qpsk_demodulator->sample_rate = sample_rate;
75 qpsk_demodulator->lo_frequency = lo_frequency;
76 qpsk_demodulator->bandwidth = bandwidth;
77 qpsk_demodulator->halfcycles_per_symbol = halfcycles_per_symbol;
78 qpsk_demodulator->carrier_start_threshold = carrier_start_threshold;
79 qpsk_demodulator->carrier_lost_threshold = carrier_lost_threshold;
80 qpsk_demodulator->pll_lock_threshold = pll_lock_threshold;
81 qpsk_demodulator->pll_lock_time = pll_lock_time;
82
83 // Use a band-pass filter if the bandwidth doesn't "touch" 0
84 if (lo_frequency - bandwidth / 2.f > 0)
85 qpsk_demodulator->rrc_filter = rrc_filter_init(FILTER_TYPE_BANDPASS,
86 sample_rate,
87 FILTER_LOBES_N,
88 FILTER_ORDER,
89 FILTER_ALPHA,
90 lo_frequency - bandwidth / 2.f,
91 lo_frequency + bandwidth / 2.f);
92
93 // Use a low-pass filter otherwise
94 else
95 qpsk_demodulator->rrc_filter = rrc_filter_init(FILTER_TYPE_LOWPASS,
96 sample_rate,
97 FILTER_LOBES_N,
98 FILTER_ORDER,
99 FILTER_ALPHA,
100 lo_frequency + bandwidth / 2.f,
101 0.f);
102
103 // Initialize PLL
104 qpsk_demodulator->pll = pll_init(sample_rate,
105 lo_frequency,
106 PLL_K_P, PLL_K_I, qpsk_demodulator->lo_frequency / 2.f,
107 halfcycles_per_symbol,
108 PLL_FREQUENCY_FILTER);
109
110 // Initialize IQ filters
111 qpsk_demodulator->butter_filter_i = butter_filter_init(FILTER_TYPE_LOWPASS,
112 sample_rate,
113 IQ_FILTER_ORDER,
114 lo_frequency * .7f,
115 sqrtf(2.f));
116 qpsk_demodulator->butter_filter_q = butter_filter_init(FILTER_TYPE_LOWPASS,
117 sample_rate,
118 IQ_FILTER_ORDER,
119 lo_frequency * .7f,
120 sqrtf(2.f));
121
122 // Initialize loop variables
123 qpsk_demodulator->sample_time = 0.f;
124 qpsk_demodulator->peak_detector = 0.f;
125 qpsk_demodulator->carrier_detected = 0;
126 qpsk_demodulator->pll_locked = 0;
127 qpsk_demodulator->pll_lock_timer = 0.f;
128 qpsk_demodulator->iq_error = -1.f;
129 qpsk_demodulator->i_sign_prev = 0.f;
130 qpsk_demodulator->q_sign_prev = 0.f;
131 for (uint8_t i = 0; i < 4; i++)
132 {
133 qpsk_demodulator->reference_signs_i[i] = 0.f;
134 qpsk_demodulator->reference_signs_q[i] = 0.f;
135 }
136 qpsk_demodulator->reference_signs_set_position = 0;
137 qpsk_demodulator->samples_after_zcd = 0;
138 qpsk_demodulator->sampling_position = -1;
139
140 return qpsk_demodulator;
141}

153void qpsk_demodulator_demodulate_chunk(qpsk_demodulator_s *qpsk_demodulator,
154 float *samples_chunk,
155 uint32_t samples_chunk_size,
156 uint8_t *symbols_chunk,
157 uint8_t debug_messages)
158{
159 // Log number of samples
160 if (debug_messages)
161 printf("Samples to demodulate: %d\n", samples_chunk_size);
162
163 // Record start time
164 clock_t time_started = clock();
165
166 for (uint32_t i = 0; i < samples_chunk_size; i++)
167 {
168 // Calculate current time (in seconds)
169 qpsk_demodulator->sample_time += 1.f / qpsk_demodulator->sample_rate;
170
171 // Extract and filter current sample
172 qpsk_demodulator->filtered_sample = rrc_filter_filter(qpsk_demodulator->rrc_filter, samples_chunk[i]);
173
174 // Calculate peak detector
175 if (fabsf(qpsk_demodulator->filtered_sample) > qpsk_demodulator->peak_detector)
176 qpsk_demodulator->peak_detector = fabsf(qpsk_demodulator->filtered_sample);
177 else
178 qpsk_demodulator->peak_detector = qpsk_demodulator->peak_detector * AGC_RELEASE_K +
179 fabsf(qpsk_demodulator->filtered_sample) * (1. - AGC_RELEASE_K);
180
181 // Check if we have any signal
182 if (qpsk_demodulator->peak_detector > 0.f)
183 {
184 // Calculate signal strength
185 qpsk_demodulator->signal_strength_rms = 20.f * log10f(qpsk_demodulator->peak_detector / sqrtf(2.f));
186
187 // Set carrier detected flag
188 if (!qpsk_demodulator->carrier_detected &&
189 qpsk_demodulator->signal_strength_rms > qpsk_demodulator->carrier_start_threshold)
190 {
191 qpsk_demodulator->carrier_detected = 1;
192 if (debug_messages)
193 printf("Carrier detected @ %d/%d (%.3f s). RMS volume: %.1f dBFS\n",
194 i + 1, samples_chunk_size,
195 qpsk_demodulator->sample_time,
196 qpsk_demodulator->signal_strength_rms);
197 }
198
199 // Carrier lost
200 if (qpsk_demodulator->carrier_detected &&
201 qpsk_demodulator->signal_strength_rms < qpsk_demodulator->carrier_lost_threshold)
202 {
203 qpsk_demodulator->carrier_detected = 0;
204 qpsk_demodulator->pll_locked = 0;
205 if (debug_messages)
206 printf("Carrier lost and PLL unlocked @ %d/%d (%.3f s). RMS volume: %.1f dBFS\n",
207 i + 1, samples_chunk_size,
208 qpsk_demodulator->sample_time,
209 qpsk_demodulator->signal_strength_rms);
210 }
211
212 // Apply automatic gain
213 if (qpsk_demodulator->carrier_detected)
214 {
215 // Apply AGC
216 qpsk_demodulator->filtered_sample *= 1.f / qpsk_demodulator->peak_detector;
217
218 // Hard-clip to -1 - 1 (to prevent huge AGC initial overshoot)
219 qpsk_demodulator->filtered_sample = clampf(qpsk_demodulator->filtered_sample, -1.f, 1.f);
220 }
221 }
222
223 // Mix input signal (real values only) with PLL real and imaginary output
224 qpsk_demodulator->i_mixed = qpsk_demodulator->filtered_sample * qpsk_demodulator->pll->nco_imag;
225 qpsk_demodulator->q_mixed = qpsk_demodulator->filtered_sample * qpsk_demodulator->pll->nco_real;
226
227 // Apply LPFs
228 qpsk_demodulator->i_mixed = butter_filter_filter(qpsk_demodulator->butter_filter_i,
229 qpsk_demodulator->i_mixed);
230 qpsk_demodulator->q_mixed = butter_filter_filter(qpsk_demodulator->butter_filter_q,
231 qpsk_demodulator->q_mixed);
232
233 // Calculate signs
234 qpsk_demodulator->i_sign = qpsk_demodulator->i_mixed >= 0.f ? 1.f : -1.f;
235 qpsk_demodulator->q_sign = qpsk_demodulator->q_mixed >= 0.f ? 1.f : -1.f;
236
237 // Check if carrier is detected but PLL still not locked
238 if (qpsk_demodulator->carrier_detected && !qpsk_demodulator->pll_locked)
239 {
240 // printf("%.2f\n", qpsk_demodulator->pll_lock_timer);
241 // Start timer
242 if (qpsk_demodulator->pll_lock_timer == 0.f)
243 qpsk_demodulator->pll_lock_timer = qpsk_demodulator->sample_time;
244
245 // Reset timer on large error or first run
246 if (fabsf(qpsk_demodulator->iq_error) > qpsk_demodulator->pll_lock_threshold)
247 qpsk_demodulator->pll_lock_timer = 0.f;
248
249 // Check timer
250 if (qpsk_demodulator->pll_lock_timer > 0.f &&
251 qpsk_demodulator->sample_time - qpsk_demodulator->pll_lock_timer > qpsk_demodulator->pll_lock_time)
252 {
253 // Reset timer
254 qpsk_demodulator->pll_lock_timer = 0.f;
255
256 // Set variables
257 qpsk_demodulator->pll_locked = 1;
258
259 qpsk_demodulator->reference_signs_i[0] = qpsk_demodulator->i_sign;
260 qpsk_demodulator->reference_signs_q[0] = qpsk_demodulator->q_sign;
261 qpsk_demodulator->reference_signs_set_position++;
262
263 // Calculated PPM error
264 float carrier_error_hz = fabsf(qpsk_demodulator->lo_frequency -
265 qpsk_demodulator->pll->frequency_filtered);
266 float carrier_error_ppm = (carrier_error_hz * 1.e6f) / qpsk_demodulator->lo_frequency;
267
268 if (debug_messages)
269 printf("PLL locked @ %d/%d (%.3f s). VCO: %.2f Hz (error between %.2f Hz: %.0f ppm). Is: %.0f, Qs: %.0f\n",
270 i + 1, samples_chunk_size, qpsk_demodulator->sample_time,
271 qpsk_demodulator->pll->frequency_filtered,
272 qpsk_demodulator->lo_frequency,
273 carrier_error_ppm,
274 qpsk_demodulator->reference_signs_i[0],
275 qpsk_demodulator->reference_signs_q[0]);
276 }
277 }
278
279 // Carrier not detected or PLL locked -> Reset timer
280 else
281 qpsk_demodulator->pll_lock_timer = 0.f;
282
283 // Zero-crossing detector (falling edge) using PLL's carrier
284 if (qpsk_demodulator->pll->zero_crossing_flag)
285 {
286 // Clear PLL's zero crossing flag
287 qpsk_demodulator->pll->zero_crossing_flag = 0;
288
289 // Reset samples counter
290 qpsk_demodulator->samples_after_zcd = 0;
291 }
292
293 // No zero-crossing detected -> Increment samples counter
294 else
295 qpsk_demodulator->samples_after_zcd++;
296
297 // Calculate how many samples one symbol takes
298 if (qpsk_demodulator->pll->frequency_filtered > 0.f)
299 qpsk_demodulator->samples_per_symbol = qpsk_demodulator->halfcycles_per_symbol *
300 (qpsk_demodulator->sample_rate /
301 (2.f * qpsk_demodulator->pll->frequency_filtered));
302 else
303 qpsk_demodulator->samples_per_symbol = 0;
304
305 // Calculate sampling position
306 if (qpsk_demodulator->carrier_detected && qpsk_demodulator->pll_locked)
307 {
308 // Calculate it only one time (at the first phase change)
309 // I know that this is not very good, and we need to do a continuous calculation (correction)
310 if (qpsk_demodulator->sampling_position < 0)
311 {
312 // If sign of I or sign of Q changed (phase change)
313 if ((qpsk_demodulator->i_sign_prev >= 0 && qpsk_demodulator->i_sign < 0) ||
314 (qpsk_demodulator->i_sign_prev < 0 && qpsk_demodulator->i_sign >= 0) ||
315 (qpsk_demodulator->q_sign_prev >= 0 && qpsk_demodulator->q_sign < 0) ||
316 (qpsk_demodulator->q_sign_prev < 0 && qpsk_demodulator->q_sign >= 0))
317 {
318 // Calculate sampling position
319 qpsk_demodulator->sampling_position = qpsk_demodulator->samples_after_zcd +
320 qpsk_demodulator->samples_per_symbol / 2.f;
321
322 // Fix if sampling position is after next zero-cross
323 if (qpsk_demodulator->sampling_position > qpsk_demodulator->samples_per_symbol)
324 qpsk_demodulator->sampling_position -= qpsk_demodulator->samples_per_symbol;
325
326 // Fix if sampling position too close to nex zero-cross
327 if (qpsk_demodulator->sampling_position > qpsk_demodulator->samples_per_symbol - 2)
328 qpsk_demodulator->sampling_position -= 2;
329
330 // Log phase change and sampling position
331 if (debug_messages)
332 printf("Detected phase change @ %d/%d (%.3f s). Sampling position: %d samples after zero-cross\n",
333 i + 1, samples_chunk_size, qpsk_demodulator->sample_time,
334 qpsk_demodulator->sampling_position);
335 }
336 }
337 }
338 // Reset sampling position
339 else
340 qpsk_demodulator->sampling_position = -1;
341
342 // Sampling position is set -> Sample data
343 if (qpsk_demodulator->sampling_position >= 0 &&
344 qpsk_demodulator->samples_after_zcd == qpsk_demodulator->sampling_position)
345 {
346 // Set other reference phases if 0 is set
347 if (0 < qpsk_demodulator->reference_signs_set_position &&
348 qpsk_demodulator->reference_signs_set_position < 4)
349 {
350 // Check if sign changed
351 if (qpsk_demodulator->i_sign !=
352 qpsk_demodulator->reference_signs_i[qpsk_demodulator->reference_signs_set_position - 1] ||
353 qpsk_demodulator->q_sign !=
354 qpsk_demodulator->reference_signs_q[qpsk_demodulator->reference_signs_set_position - 1])
355 {
356 // Record new reference signs
357 qpsk_demodulator->reference_signs_i[qpsk_demodulator->reference_signs_set_position] =
358 qpsk_demodulator->i_sign;
359 qpsk_demodulator->reference_signs_q[qpsk_demodulator->reference_signs_set_position] =
360 qpsk_demodulator->q_sign;
361 if (debug_messages)
362 printf("IQ values of phase %d recorded @ %d/%d (%.3f s). Is: %.0f, Qs: %.0f\n",
363 qpsk_demodulator->reference_signs_set_position,
364 i + 1, samples_chunk_size, qpsk_demodulator->sample_time,
365 qpsk_demodulator->i_sign, qpsk_demodulator->q_sign);
366 qpsk_demodulator->reference_signs_set_position++;
367 }
368
369 // Save empty symbol
370 symbols_chunk[i] = 0xFF;
371 }
372
373 // Check if we have all reference values
374 else if (qpsk_demodulator->reference_signs_set_position >= 4)
375 {
376 // Decode data according to table of reference signs
377 uint8_t iq_symbol = 0;
378 for (uint8_t sign_n = 0; sign_n < 4; sign_n++)
379 {
380 if (qpsk_demodulator->reference_signs_i[sign_n] == qpsk_demodulator->i_sign &&
381 qpsk_demodulator->reference_signs_q[sign_n] == qpsk_demodulator->q_sign)
382 {
383 iq_symbol = sign_n;
384 break;
385 }
386 }
387
388 // Save decoded symbol
389 symbols_chunk[i] = iq_symbol;
390
391 // Log decoded symbol
392 if (debug_messages)
393 printf("Decoded symbol @ %d/%d (%.3f s): %d (0b" BYTE_TO_BINARY_PATTERN "). I: %.2f, Q: %.2f\n",
394 i + 1, samples_chunk_size, qpsk_demodulator->sample_time,
395 iq_symbol,
396 BYTE_TO_BINARY(iq_symbol),
397 qpsk_demodulator->i_mixed, qpsk_demodulator->q_mixed);
398 }
399 // Save empty symbol
400 else
401 symbols_chunk[i] = 0xFF;
402 }
403 // Save empty symbol
404 else
405 symbols_chunk[i] = 0xFF;
406
407 // Save signs for the next cycle
408 qpsk_demodulator->i_sign_prev = qpsk_demodulator->i_sign;
409 qpsk_demodulator->q_sign_prev = qpsk_demodulator->q_sign;
410
411 // Reset signs on carrier lost
412 if (!qpsk_demodulator->pll_locked || !qpsk_demodulator->carrier_detected)
413 {
414 qpsk_demodulator->reference_signs_set_position = 0;
415 for (uint8_t sign_n = 0; sign_n < 4; sign_n++)
416 {
417 qpsk_demodulator->reference_signs_i[sign_n] = 0.f;
418 qpsk_demodulator->reference_signs_q[sign_n] = 0.f;
419 }
420 }
421
422 // Calculate PLL input
423 float pll_input_i = qpsk_demodulator->i_sign * qpsk_demodulator->q_mixed;
424 float pll_input_q = qpsk_demodulator->q_sign * qpsk_demodulator->i_mixed;
425
426 // Calculate IQ error
427 qpsk_demodulator->iq_error = qpsk_demodulator->carrier_detected ? (pll_input_q - pll_input_i) : 0.f;
428
429 // Take one PLL step
430 pll_step(qpsk_demodulator->pll, qpsk_demodulator->iq_error, qpsk_demodulator->sample_time);
431 }
432
433 if (debug_messages)
434 {
435 // Record stop time
436 clock_t time_stopped = clock();
437
438 // Calculate time
439 double time_taken_millis = (((double)(time_stopped - time_started)) / CLOCKS_PER_SEC) * 1000.;
440 printf("Demodulation is complete. Time taken: ~%.0f ms\n", time_taken_millis);
441 }
