84void pll_step(pll_s *pll, float error, float sample_time)
85{
86 // Calculate integral controller
87 pll->integrator += pll->k_i * error * (1.f / pll->sample_rate);
88
89 // Calculate PI output using proportional controller and integral controller
90 pll->pi_out = pll->k_p * error + pll->integrator;
91
92 // Calculate output phase
93 pll->output_phase += 2.f * M_PI * pll->pi_out * pll->k_nco * (1.f / pll->sample_rate);
94
95 // Calculate base for sin and cos
96 pll->omega_prev = pll->omega;
97 pll->omega = 2.f * M_PI * pll->lo_frequency * sample_time + pll->output_phase;
98 pll->omega = fmodf(pll->omega, 2.f * M_PI);
99
100 // Calculate momentary phase difference
101 if (pll->omega > pll->omega_prev)
102 pll->omega_diff = pll->omega - pll->omega_prev;
103 else
104 pll->omega_diff = (pll->omega + (2.f * M_PI)) - pll->omega_prev;
105
106 // Calculate current VCO frequency
107 pll->frequency = (pll->omega_diff / (2.f * M_PI)) * pll->sample_rate;
108 if (pll->frequency > 0.f)
109 pll->frequency_filtered = pll->frequency_filtered * pll->frequency_filter_k +
110 pll->frequency * (1.f - pll->frequency_filter_k);
111
112 // Half-cycle detector
113 // omega_prev <= pi <= omega and omega != omega_prev -> pi cross
114 // omega_prev > omega -> 2*pi cross
115 if ((pll->omega_prev <= M_PI && M_PI <= pll->omega && pll->omega != pll->omega_prev) || pll->omega_prev > pll->omega)
116 pll->halfcycles_counter += 1;
117
118 // Symbol rate detector (aka Zero-crossing detector) @ halfcycles_per_symbol * PI (symbol rate)
119 if (pll->halfcycles_counter >= pll->halfcycles_per_symbol)
120 {
121 pll->halfcycles_counter = 0;
122 pll->zero_crossing_flag = 1;
123 }
124
125 // Calculate complex output
126 pll->nco_real = cosf(pll->omega);
127 pll->nco_imag = -sinf(pll->omega);
128}
