44rrc_filter_s *rrc_filter_init(uint8_t pass_type,
45 float sample_rate,
46 uint8_t positive_lobes_n,
47 uint8_t order,
48 float alpha,
49 float cutoff_frequency,
50 float cutoff_frequency_2)
51{
52 // Allocate struct
53 rrc_filter_s *filter = malloc(sizeof(rrc_filter_s));
54
55 // Initialize basic fields
56 filter->pass_type = pass_type;
57 filter->sample_rate = sample_rate;
58 filter->positive_lobes_n = positive_lobes_n;
59 filter->order = order;
60 filter->alpha = alpha;
61 filter->cutoff_frequency = cutoff_frequency;
62 filter->cutoff_frequency_2 = cutoff_frequency_2;
63
64 // Calculate filter length
65 if (filter->pass_type == FILTER_TYPE_BANDPASS)
66 filter->filter_length = (uint32_t)(filter->sample_rate /
67 (2.f * ((filter->cutoff_frequency + filter->cutoff_frequency_2) / 2.f))) *
68 ((uint32_t)filter->positive_lobes_n * 2 + 1);
69 else
70 filter->filter_length = (uint32_t)(filter->sample_rate / (2.f * filter->cutoff_frequency)) *
71 ((uint32_t)filter->positive_lobes_n * 2.f + 1.f);
72
73 // Allocate and initialize filter_state (for future convolution)
74 filter->filter_state = malloc(order * sizeof(float *));
75 for (uint8_t i = 0; i < order; i++)
76 {
77 filter->filter_state[i] = malloc(filter->filter_length * sizeof(float));
78 for (uint32_t j = 0; j < filter->filter_length; j++)
79 filter->filter_state[i][j] = 0.f;
80 }
81
82 // Generate low-pass impulse response
83 filter->impulse_response = _rrc_filter_create_ir(filter, filter->cutoff_frequency);
84
85 // Calculate impulse response for high-pass filter
86 if (filter->pass_type == FILTER_TYPE_HIGHPASS)
87 {
88 // Create delta function
89 float *delta_function = malloc(filter->filter_length * sizeof(float));
90 for (uint32_t i = 0; i < filter->filter_length; i++)
91 if (i == filter->filter_length / 2)
92 delta_function[i] = 1.f;
93 else
94 delta_function[i] = 0.f;
95
96 // Subtract to make impulse response high-pass
97 for (uint32_t i = 0; i < filter->filter_length; i++)
98 filter->impulse_response[i] = delta_function[i] - filter->impulse_response[i];
99
100 // Delete delta function
101 free(delta_function);
102 }
103
104 // Calculate impulse response for band-pass filter
105 else if (filter->pass_type == FILTER_TYPE_BANDPASS)
106 {
107 // Generate low-pass impulse response for high-pass component
108 float *impulse_response_upper = _rrc_filter_create_ir(filter, filter->cutoff_frequency_2);
109
110 // Subtract upper impulse response from lower to make it band-pass
111 for (uint32_t i = 0; i < filter->filter_length; i++)
112 filter->impulse_response[i] = filter->impulse_response[i] - impulse_response_upper[i];
113
114 // Delete upper impulse response
115 free(impulse_response_upper);
116 }
117
118 return filter;
119}

128float *_rrc_filter_create_ir(rrc_filter_s *filter, float cutoff_frequency)
129{
130 // Calculate Ts
131 // P.S. cutoff_frequency * filter->alpha * sqrtf(filter->alpha) is a compensation for alpha (added experimentally)
132 float ts = 1.f / (cutoff_frequency + cutoff_frequency * filter->alpha * sqrtf(filter->alpha));
133
134 // Invert alpha to calculate beta
135 float beta = 1.f - filter->alpha;
136
137 // Allocate memory
138 float *impulse_response = malloc(filter->filter_length * sizeof(float));
139
140 float t;
141 float term_0, term_1, term_2, term_3, term_4;
142 for (uint32_t i = 0; i < filter->filter_length; i++)
143 {
144 // Make "virtual" 0 at the center
145 t = (-(float)filter->filter_length / 2.f + i) / filter->sample_rate;
146
147 // t = 0
148 if (t == 0.)
149 {
150 term_0 = 1.f / ts;
151 term_1 = (1.f + beta * (4.f / M_PI - 1.f));
152 impulse_response[i] = term_0 * term_1 / filter->sample_rate;
153 }
154
155 // t = +/- ts/(4 * beta)
156 else if (abs(t) == (ts / (4.f * beta)))
157 {
158 term_0 = (beta / (ts * sqrtf(2.f)));
159 term_1 = (1.f + 2.f / M_PI);
160 term_2 = sinf(M_PI / (4.f * beta));
161 term_3 = (1.f - 2.f / M_PI);
162 term_4 = cosf(M_PI / (4.f * beta));
163 impulse_response[i] = term_0 * (term_1 * term_2 + term_3 * term_4) / filter->sample_rate;
164 }
165
166 // Otherwise
167 else
168 {
169 term_0 = 1.f / ts;
170 term_1 = sinf(M_PI * (t / ts) * (1.f - beta));
171 term_2 = 4.f * beta * (t / ts) * cosf(M_PI * (t / ts) * (1.f + beta));
172 term_3 = M_PI * (t / ts);
173 term_4 = 1.f - powf(4.f * beta * (t / ts), 2.f);
174 impulse_response[i] = term_0 * (term_1 + term_2) / (term_3 * term_4) / filter->sample_rate;
175 }
176 }
177
178 return impulse_response;
179}

188float rrc_filter_filter(rrc_filter_s *filter, float input_value)
189{
190 // Filter same value multiple times
191 filter->filtered_value_temp = input_value;
192 for (uint8_t order_i = 0; order_i < filter->order; order_i++)
193 {
194 // Fill rotating buffer
195 for (uint32_t i = 0; i < filter->filter_length - 1; i++)
196 filter->filter_state[order_i][i] = filter->filter_state[order_i][i + 1];
197 filter->filter_state[order_i][filter->filter_length - 1] = filter->filtered_value_temp;
198
199 // Convolve
200 filter->output_temp = 0.f;
201 for (uint32_t i = 0; i < filter->filter_length; i++)
202 filter->output_temp += filter->filter_state[order_i][i] * filter->impulse_response[i];
203
204 // Save for next order
205 filter->filtered_value_temp = filter->output_temp;
206 }
207
208 return filter->filtered_value_temp;
209}
