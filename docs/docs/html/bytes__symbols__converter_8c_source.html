78 start_silence_length,

83 // Fill start silence
84 for (uint32_t i = 0; i < start_silence_length; i++)
85 symbols[i] = 0xFF;
86
87 // Fill preamble
88 for (uint32_t i = 0; i < preamble_length; i++)
89 symbols[start_silence_length + i] = 0;
90
91 // Add IQ (initial phase) sync byte (0b00011011) (0 1 2 3)
92 if (add_iq_sync_byte)
93 {
94 symbols[start_silence_length + preamble_length] = 0b00;
95 symbols[start_silence_length + preamble_length + 1] = 0b01;
96 symbols[start_silence_length + preamble_length + 2] = 0b10;
97 symbols[start_silence_length + preamble_length + 3] = 0b11;
98 }
99
100 // Fill bytes
101 uint32_t symbols_counter = 0;
102 uint8_t current_byte = 0;
103 for (uint32_t i = 0; i < bytes_length * 4; i++)
104 {
105 // Extract new byte and reset symbols (bit) counter
106 if (symbols_counter == 0)
107 {
108 current_byte = bytes[i / 4];
109 symbols_counter = 4;
110 }
111 symbols_counter--;
112
113 // Fill symbols arrays
114 symbols[start_silence_length + preamble_length + (add_iq_sync_byte ? 4 : 0) + i] =
115 (current_byte >> (symbols_counter * 2)) & 0b11;
116 }
117
118 // Fill end silence
119 for (uint32_t i = 0; i < end_silence_length; i++)
120 symbols[start_silence_length + preamble_length + (add_iq_sync_byte ? 4 : 0) + bytes_length * 4 + i] = 0xFF;

132uint8_t *symbols_to_bytes(uint8_t *symbols, uint32_t symbols_length)
133{
134 // Allocate array of bytes
135 uint8_t *bytes = malloc(symbols_length / 4 * sizeof(uint8_t));
136
137 uint8_t current_byte = 0;
138 for (uint32_t i = 0; i < symbols_length; i++)
139 {
140 // Build from symbol bits
141 current_byte |= (symbols[i] & 0b11) << (6 - (i % 4) * 2);
142 if (i % 4 == 3)
143 {
144 bytes[i / 4] = current_byte;
145 current_byte = 0;
146 }
147 }
148
149 return bytes;
150}
