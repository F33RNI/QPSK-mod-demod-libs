44qpsk_modulator_s *qpsk_modulator_init(float sample_rate,
45 float carrier_frequency,
46 float bandwidth,
47 uint16_t halfcycles_per_symbol,
48 float amplitude_peak)
49{
50 // Allocate struct
51 qpsk_modulator_s *qpsk_modulator = malloc(sizeof(qpsk_modulator_s));
52
53 // Initialize basic fields
54 qpsk_modulator->sample_rate = sample_rate;
55 qpsk_modulator->carrier_frequency = carrier_frequency;
56 qpsk_modulator->bandwidth = bandwidth;
57 qpsk_modulator->halfcycles_per_symbol = halfcycles_per_symbol;
58 qpsk_modulator->amplitude_peak = amplitude_peak;
59
60 // Calculate how much samples one symbol takes
61 qpsk_modulator->samples_per_symbol = (uint32_t)((1.f / qpsk_modulator->carrier_frequency / 2.f) *
62 qpsk_modulator->sample_rate *
63 (float)qpsk_modulator->halfcycles_per_symbol);
64
65 // Use a band-pass filter if the bandwidth doesn't "touch" 0
66 if (carrier_frequency - bandwidth / 2.f > 0)
67 qpsk_modulator->rrc_filter = rrc_filter_init(FILTER_TYPE_BANDPASS,
68 sample_rate,
69 FILTER_LOBES_N,
70 FILTER_ORDER,
71 FILTER_ALPHA,
72 carrier_frequency - bandwidth / 2.f,
73 carrier_frequency + bandwidth / 2.f);
74
75 // Use a low-pass filter otherwise
76 else
77 qpsk_modulator->rrc_filter = rrc_filter_init(FILTER_TYPE_LOWPASS,
78 sample_rate,
79 FILTER_LOBES_N,
80 FILTER_ORDER,
81 FILTER_ALPHA,
82 carrier_frequency + bandwidth / 2.f,
83 0.f);
84
85 // Initialize loop variables
86 qpsk_modulator->samples_per_symbol_counter = 0;
87 qpsk_modulator->sample_counter = 0;
88 qpsk_modulator->symbol_counter = 0;
89 qpsk_modulator->sample_time = 0.f;
90 qpsk_modulator->i_sign = 0.f;
91 qpsk_modulator->q_sign = 0.f;
92 qpsk_modulator->peak_detector = 0.f;
93
94 return qpsk_modulator;
95}

120void qpsk_modulator_modulate_chunk(qpsk_modulator_s *qpsk_modulator,
121 uint8_t *symbols_chunk,
122 uint32_t symbols_chunk_length,
123 float *samples_chunk,
124 uint8_t debug_messages)
125{
126 // Log number of samples
127 if (debug_messages)
128 printf("Symbols to modulate: %d\n", symbols_chunk_length);
129
130 // Record start time
131 clock_t time_started = clock();
132
133 // Modulate entire chunk
134 while (1)
135 {
136 // Calculate current time (in seconds)
137 qpsk_modulator->sample_time += 1.f / qpsk_modulator->sample_rate;
138
139 // New symbol
140 if (qpsk_modulator->samples_per_symbol_counter == 0)
141 {
142 // Check if we have data to modulate
143 if (qpsk_modulator->symbol_counter <= symbols_chunk_length - 1)
144 {
145 // Extract symbol
146 qpsk_modulator->current_symbol = symbols_chunk[qpsk_modulator->symbol_counter];
147
148 // 00, 01, 10 or 11
149 if (qpsk_modulator->current_symbol <= 0b11)
150 {
151 qpsk_modulator->i_sign = qpsk_modulator->current_symbol & 0b10 ? 1.f : -1.f;
152 qpsk_modulator->q_sign = qpsk_modulator->current_symbol & 0b01 ? 1.f : -1.f;
153 }
154
155 // Anything else (0xFF)
156 else
157 {
158 qpsk_modulator->i_sign = 0.f;
159 qpsk_modulator->q_sign = 0.f;
160 }
161
162 // "Play" new symbol for some amount of cycles
163 qpsk_modulator->samples_per_symbol_counter = qpsk_modulator->samples_per_symbol;
164
165 // Log
166 if (debug_messages)
167 printf("Modulating %d/%d symbol: %d (0b" BYTE_TO_BINARY_PATTERN "). I: %.0f, Q: %.0f\n",
168 qpsk_modulator->symbol_counter + 1, symbols_chunk_length,
169 qpsk_modulator->current_symbol, BYTE_TO_BINARY(qpsk_modulator->current_symbol),
170 qpsk_modulator->i_sign, qpsk_modulator->q_sign);
171
172 // Increment counter
173 qpsk_modulator->symbol_counter++;
174 }
175
176 // Exit if no more data
177 else
178 break;
179 }
180
181 // Decrement samples counter
182 if (qpsk_modulator->samples_per_symbol_counter > 0)
183 qpsk_modulator->samples_per_symbol_counter--;
184
185 // Calculate IQ sample
186 qpsk_modulator->iq_sample = cosf(2.f * M_PI * qpsk_modulator->carrier_frequency *
187 qpsk_modulator->sample_time) *
188 qpsk_modulator->i_sign * AMPLITUDE_CORRECTION_FACTOR;
189 qpsk_modulator->iq_sample -= sinf(2.f * M_PI * qpsk_modulator->carrier_frequency *
190 qpsk_modulator->sample_time) *
191 qpsk_modulator->q_sign * AMPLITUDE_CORRECTION_FACTOR;
192
193 // Filter sample
194 qpsk_modulator->iq_sample = rrc_filter_filter(qpsk_modulator->rrc_filter, qpsk_modulator->iq_sample);
195
196 // Calculate peak detector
197 if (fabsf(qpsk_modulator->iq_sample) > qpsk_modulator->peak_detector)
198 qpsk_modulator->peak_detector = fabsf(qpsk_modulator->iq_sample);
199 else
200 qpsk_modulator->peak_detector = qpsk_modulator->peak_detector * AGC_RELEASE_K +
201 fabsf(qpsk_modulator->iq_sample) * (1. - AGC_RELEASE_K);
202
203 // Apply AGC and ignore silence
204 if (qpsk_modulator->peak_detector > AGC_APPLY_THRESHOLD)
205 qpsk_modulator->iq_sample *= qpsk_modulator->amplitude_peak / qpsk_modulator->peak_detector;
206 else
207 qpsk_modulator->iq_sample *= qpsk_modulator->amplitude_peak;
208
209 // Write to the array
210 samples_chunk[qpsk_modulator->sample_counter] = qpsk_modulator->iq_sample;
211
212 // Increment chunk array counter
213 qpsk_modulator->sample_counter++;
214 }
215
216 if (debug_messages)
217 {
218 // Record stop time
219 clock_t time_stopped = clock();
220
221 // Calculate time
222 double time_taken_millis = (((double)(time_stopped - time_started)) / CLOCKS_PER_SEC) * 1000.;
223 printf("Modulation is complete. Time taken: ~%.0f ms\n", time_taken_millis);
224 }
225}
